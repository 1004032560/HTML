## 3、组件切换

### 3.1、方式一

~~~html
<body>
    <div id="app">
        <login v-if="flag"></login>
        <register v-if="!flag"></register>
        <button @click="login">登录</button>
        <button @click="register">注册</button>
    </div>
    <script type="text/javascript">
        var vm = new Vue({
            el:'#app',
            data:{
                msg:'hello',
                flag:true
            },
            methods:{
                login(){
                    this.flag = true
                },
                register(){
                    this.flag = false
                }
            },
            components:{
                'login':{
                    template:'<h1>登录</h1>'
                },
                'register':{
                    template:'<h1>注册</h1>'
                }
            }
        })
    </script>
</body>
~~~

### 3.2、方式二

~~~html

        <button @click="comName='login'">登录</button>

    </div>
    <script type="text/javascript">
        var vm = new Vue({
            el:'#app',
            data:{
                msg:'hello',
                comName:''
            },

                    this.flag = true

                },

                }
            }
        })
    </script>
</body>
~~~
